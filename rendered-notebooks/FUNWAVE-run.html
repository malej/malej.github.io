def readStationData(num,numOfStations,postprocessDir):
    """Function that reads in the enrie sta_00.. ASCII file and returns
    it in the 2D NumPy array of size [numOfRows,numOfColums]."""
    assert num in range(1,numOfStations+1), "File Index:%d is not in range of station numbers." %(num,)
    if num < 10:
        fileIndex = '0'+str(num)
        fileName = postprocessDir+'/sta_00{0:s}'.format(fileIndex)
    else:
        fileIndex = str(num)
        fileName = postprocessDir+'/sta_00{0:s}'.format(fileIndex)

    fin = open(fileName, "r")
    print "READING IN:", "sta_00{0:s} ... ".format(fileIndex)
    lines = fin.readlines()
    numpyFileArray = np.zeros((len(lines),4))
    for lineIndex,line in enumerate(lines):
        rowDataList = line.strip().split() # dropping all whitespace (leading & trailing)
        numpyData = np.array(rowDataList).astype('float')
        numpyFileArray[lineIndex,:] = numpyData[:]
    fin.close()
    return numpyFileArray

def readMketaData(postprocessDir,columnsMketa):
    """Function that reads in the enrie mketa.out ASCII file and returns
    it in the 2D NumPy array of size [rowsMketa,columnsMketa]."""
    fileName = postprocessDir+'/mketa.out'
    fin = open(fileName,"r")
    print "READING IN:", fin.name
    lines = fin.readlines()
    numpyFileArray = np.zeros((len(lines),columnsMketa))
    for lineIndex, line in enumerate(lines):
        rowData = line.strip().split() # droping white spaces
        numpyData = np.array(rowData).astype('float')
        numpyFileArray[lineIndex,:] = numpyData
    fin.close
    return numpyFileArray


# Process FILEs: read in data in sta_00.. ASCII files
for station in range(1,numOfStations+1):
